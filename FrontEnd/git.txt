document.addEventListener('DOMContentLoaded', function () {
    // Fetch and display GitHub repositories
    fetch('https://api.github.com/repos/MozeSiame/GROUP-11/contents/ASSIGNMENT 2/index.html')
      .then(response => response.json())
      .then(data => {
        const projectGallery = document.getElementById('project-gallery');
        data.forEach(repo => {
          const projectDiv = document.createElement('div');
          projectDiv.classList.add('project');
          projectDiv.innerHTML = `
            <h3>Attendance_Register</h3>
            <p>An attendance register project</p>
            <a href="https://github.com/MozeSiame/GROUP-11/blob/main/ASSIGNMENT%202/index.html" target="_blank">View on GitHub</a>
          `;
          projectGallery.appendChild(projectDiv);
        });
      })
      .catch(error => console.error('Error fetching GitHub repos:', error));
